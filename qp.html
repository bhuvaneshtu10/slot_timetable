Write a shell script to check if the Nginx web server is running using ladder if-else-if.

#!/bin/bash
# nginx-check.sh - Example using if-elif-else ladder

if systemctl is-active --quiet nginx; then
  echo "Nginx is running."
elif pgrep -x nginx >/dev/null 2>&1; then
  echo "Nginx process found but service not active (maybe started manually)."
else
  echo "Nginx is not running."
  # optional: start nginx
  # sudo systemctl start nginx
fi
